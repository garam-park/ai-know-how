## What (무엇을)
/projects/:id/dashboard에서 프로젝트 현황을 한눈에 파악할 수 있는 대시보드를 제공합니다. 전체 진척률, D-Day, 멤버 수, WBS 현황, 회사별 진척률, 상태별 파이차트 등을 포함합니다.

## Why (왜)
프로젝트의 전체 건강도를 시각적으로 파악하는 것이 프로젝트 관리의 핵심입니다.

### Step 1: 대시보드 API 서비스 생성
`apps/frontend/src/services/dashboardService.ts` 파일을 생성합니다:

```typescript
import apiClient from './api';

export interface DashboardData {
  projectId: string;
  projectName: string;
  overallProgress: number;
  startDate: string;
  endDate: string;
  daysLeft: number;
  memberCount: number;
  wbsStats: {
    total: number;
    completed: number;
    inProgress: number;
    notStarted: number;
  };
  companyProgress: Array<{
    companyId: string;
    companyName: string;
    progress: number;
    memberCount: number;
  }>;
  statusDistribution: {
    completed: number;
    inProgress: number;
    notStarted: number;
  };
}

export const dashboardService = {
  getDashboardData: async (projectId: string): Promise<DashboardData> => {
    const response = await apiClient.get(`/projects/${projectId}/dashboard`);
    return response.data;
  },
};
```

### Step 3: DashboardPage 컴포넌트 생성
`apps/frontend/src/pages/projects/DashboardPage.tsx` 파일을 생성합니다:

```typescript
import React, { useEffect, useState } from 'react';
import {
  LineChart,
  Line,
  PieChart,
  Pie,
  Cell,
  BarChart,
  Bar,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  Legend,
  ResponsiveContainer,
} from 'recharts';
import { useProjectContext } from '../../hooks/useProjectContext';
import { dashboardService, DashboardData } from '../../services/dashboardService';
import Card from '../../components/Card';
import Loading from '../../components/Loading';
import './DashboardPage.css';

const DashboardPage: React.FC = () => {
  const { project } = useProjectContext();
  const [dashboardData, setDashboardData] = useState<DashboardData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetchDashboard();
  }, [project.id]);

  const fetchDashboard = async () => {
    try {
      setLoading(true);
      const data = await dashboardService.getDashboardData(project.id);
      setDashboardData(data);
    } catch (err) {
      setError(
        err instanceof Error ? err.message : '대시보드 데이터를 불러올 수 없습니다'
      );
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return <Loading message="대시보드 로딩 중..." />;
  }

  if (error || !dashboardData) {
    return <div className="dashboard__error">{error || '대시보드를 불러올 수 없습니다'}</div>;
  }

  const daysLeftClass = dashboardData.daysLeft < 0 ? 'danger' : '';
  const progressPercent = dashboardData.overallProgress;

  const pieData = [
    { name: '완료', value: dashboardData.statusDistribution.completed },
    { name: '진행중', value: dashboardData.statusDistribution.inProgress },
    { name: '미시작', value: dashboardData.statusDistribution.notStarted },
  ];

  const colors = ['#4caf50', '#2196f3', '#ccc'];

  return (
    <div className="dashboard">
      <h2 className="dashboard__title">프로젝트 대시보드</h2>

      {/* KPI 카드들 */}
      <div className="dashboard__kpi-grid">
        <Card className="dashboard__kpi-card">
          <div className="dashboard__kpi-content">
            <h3>전체 진척도</h3>
            <div className="dashboard__progress-circle">
              <svg viewBox="0 0 100 100" className="dashboard__progress-svg">
                <circle cx="50" cy="50" r="45" className="dashboard__progress-bg" />
                <circle
                  cx="50"
                  cy="50"
                  r="45"
                  className="dashboard__progress-fill"
                  style={{
                    strokeDashoffset: `${282.7 * (1 - progressPercent / 100)}`,
                  }}
                />
                <text x="50" y="50" textAnchor="middle" dy="0.3em">
                  {progressPercent}%
                </text>
              </svg>
            </div>
          </div>
        </Card>

        <Card className="dashboard__kpi-card">
          <div className="dashboard__kpi-content">
            <h3>D-Day</h3>
            <div className={`dashboard__kpi-value ${daysLeftClass}`}>
              D{dashboardData.daysLeft >= 0 ? '-' : ''}{Math.abs(dashboardData.daysLeft)}
            </div>
            <p className="dashboard__kpi-label">
              {new Date(dashboardData.endDate).toLocaleDateString('ko-KR')}
            </p>
          </div>
        </Card>

        <Card className="dashboard__kpi-card">
          <div className="dashboard__kpi-content">
            <h3>참여 멤버</h3>
            <div className="dashboard__kpi-value">{dashboardData.memberCount}</div>
            <p className="dashboard__kpi-label">명</p>
          </div>
        </Card>

        <Card className="dashboard__kpi-card">
          <div className="dashboard__kpi-content">
            <h3>총 작업</h3>
            <div className="dashboard__kpi-value">{dashboardData.wbsStats.total}</div>
            <p className="dashboard__kpi-label">개</p>
          </div>
        </Card>
      </div>

      {/* 차트들 */}
      <div className="dashboard__charts-grid">
        <Card className="dashboard__chart-card" title="WBS 상태 분포">
          <ResponsiveContainer width="100%" height={300}>
            <PieChart>
              <Pie
                data={pieData}
                cx="50%"
                cy="50%"
                labelLine={false}
                label={({ name, value }) => `${name}: ${value}`}
                outerRadius={100}
                fill="#8884d8"
                dataKey="value"
              >
                {colors.map((color, index) => (
                  <Cell key={`cell-${index}`} fill={color} />
                ))}
              </Pie>
              <Tooltip />
            </PieChart>
          </ResponsiveContainer>
        </Card>

        <Card className="dashboard__chart-card" title="회사별 진척률">
          <ResponsiveContainer width="100%" height={300}>
            <BarChart data={dashboardData.companyProgress}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis
                dataKey="companyName"
                angle={-45}
                textAnchor="end"
                height={100}
              />
              <YAxis />
              <Tooltip />
              <Bar dataKey="progress" fill="#667eea" />
            </BarChart>
          </ResponsiveContainer>
        </Card>

        <Card className="dashboard__chart-card dashboard__chart-card--full" title="WBS 진행 현황">
          <div className="dashboard__wbs-stats">
            <div className="dashboard__wbs-stat">
              <div className="dashboard__stat-icon completed">✓</div>
              <div>
                <p className="dashboard__stat-label">완료</p>
                <p className="dashboard__stat-value">
                  {dashboardData.wbsStats.completed}
                </p>
              </div>
            </div>
            <div className="dashboard__wbs-stat">
              <div className="dashboard__stat-icon inprogress">⟳</div>
              <div>
                <p className="dashboard__stat-label">진행중</p>
                <p className="dashboard__stat-value">
                  {dashboardData.wbsStats.inProgress}
                </p>
              </div>
            </div>
            <div className="dashboard__wbs-stat">
              <div className="dashboard__stat-icon notstarted">◯</div>
              <div>
                <p className="dashboard__stat-label">미시작</p>
                <p className="dashboard__stat-value">
                  {dashboardData.wbsStats.notStarted}
                </p>
              </div>
            </div>
          </div>
        </Card>

        <Card className="dashboard__chart-card dashboard__chart-card--full" title="참여 회사">
          <div className="dashboard__company-list">
            {dashboardData.companyProgress.map((company, index) => (
              <div key={index} className="dashboard__company-item">
                <span className="dashboard__company-name">{company.companyName}</span>
                <span className="dashboard__company-members">
                  멤버: {company.memberCount}명
                </span>
                <div className="dashboard__company-progress-bar">
                  <div
                    className="dashboard__company-progress-fill"
                    style={{ width: `${company.progress}%` }}
                  />
                </div>
                <span className="dashboard__company-progress-text">
                  {company.progress}%
                </span>
              </div>
            ))}
          </div>
        </Card>
      </div>
    </div>
  );
};

export default DashboardPage;
```
