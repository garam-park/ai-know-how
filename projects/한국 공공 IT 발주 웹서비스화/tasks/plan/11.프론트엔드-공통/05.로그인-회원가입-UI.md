# 로그인·회원가입 UI

## 예상 소요시간
| 구분 | 시간 |
|------|------|
| 사람 (숙련 개발자) | ~40분 |
| AI 에이전트 | ~15분 |

## What (무엇을)
로그인 페이지와 회원가입 페이지를 구현합니다. 폼 검증, 로딩 상태 표시, 에러 메시지 표시를 포함하며, 성공 시 /projects로 자동 리다이렉트됩니다.

## Why (왜)
사용자 인증의 진입점입니다. 직관적인 UI와 명확한 에러 메시지는 사용자 경험을 크게 향상시킵니다.

## 선행 작업
- `11.프론트엔드-공통/04.인증-상태관리-Context.md`

## 수행 가이드

### Step 1: LoginPage 컴포넌트 생성
`apps/frontend/src/pages/auth/LoginPage.tsx` 파일을 생성합니다:

```typescript
import React, { useState } from 'react';
import { useNavigate, Link } from 'react-router-dom';
import { useAuth } from '../../hooks/useAuth';
import './AuthPage.css';

const LoginPage: React.FC = () => {
  const navigate = useNavigate();
  const { login, isLoading, error } = useAuth();

  const [formData, setFormData] = useState({
    email: '',
    password: '',
  });

  const [formError, setFormError] = useState<Record<string, string>>({});

  const validateForm = (): boolean => {
    const errors: Record<string, string> = {};

    if (!formData.email) {
      errors.email = '이메일은 필수입니다';
    } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(formData.email)) {
      errors.email = '올바른 이메일 형식이 아닙니다';
    }

    if (!formData.password) {
      errors.password = '비밀번호는 필수입니다';
    } else if (formData.password.length < 6) {
      errors.password = '비밀번호는 최소 6자 이상이어야 합니다';
    }

    setFormError(errors);
    return Object.keys(errors).length === 0;
  };

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;
    setFormData((prev) => ({
      ...prev,
      [name]: value,
    }));
    // 입력 시 에러 메시지 제거
    if (formError[name]) {
      setFormError((prev) => ({
        ...prev,
        [name]: '',
      }));
    }
  };

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();

    if (!validateForm()) {
      return;
    }

    try {
      await login(formData.email, formData.password);
      navigate('/projects');
    } catch (err) {
      console.error('로그인 오류:', err);
      // 에러는 useAuth의 error 상태로 표시됨
    }
  };

  return (
    <div className="auth-page">
      <div className="auth-form">
        <h1 className="auth-form__title">로그인</h1>

        {error && <div className="auth-form__error">{error}</div>}

        <form onSubmit={handleSubmit} className="auth-form__form">
          <div className="auth-form__field">
            <label htmlFor="email" className="auth-form__label">
              이메일
            </label>
            <input
              type="email"
              id="email"
              name="email"
              value={formData.email}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.email ? 'auth-form__input--error' : ''
              }`}
              placeholder="user@example.com"
              disabled={isLoading}
            />
            {formError.email && (
              <span className="auth-form__error-message">{formError.email}</span>
            )}
          </div>

          <div className="auth-form__field">
            <label htmlFor="password" className="auth-form__label">
              비밀번호
            </label>
            <input
              type="password"
              id="password"
              name="password"
              value={formData.password}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.password ? 'auth-form__input--error' : ''
              }`}
              placeholder="••••••"
              disabled={isLoading}
            />
            {formError.password && (
              <span className="auth-form__error-message">
                {formError.password}
              </span>
            )}
          </div>

          <button
            type="submit"
            className="auth-form__button"
            disabled={isLoading}
          >
            {isLoading ? '로그인 중...' : '로그인'}
          </button>
        </form>

        <div className="auth-form__footer">
          <p>
            계정이 없으신가요?{' '}
            <Link to="/register" className="auth-form__link">
              회원가입
            </Link>
          </p>
        </div>
      </div>
    </div>
  );
};

export default LoginPage;
```

### Step 2: RegisterPage 컴포넌트 생성
`apps/frontend/src/pages/auth/RegisterPage.tsx` 파일을 생성합니다:

```typescript
import React, { useState } from 'react';
import { useNavigate, Link } from 'react-router-dom';
import { useAuth } from '../../hooks/useAuth';
import './AuthPage.css';

const RegisterPage: React.FC = () => {
  const navigate = useNavigate();
  const { register, isLoading, error } = useAuth();

  const [formData, setFormData] = useState({
    email: '',
    password: '',
    confirmPassword: '',
    name: '',
    tel: '',
  });

  const [formError, setFormError] = useState<Record<string, string>>({});

  const validateForm = (): boolean => {
    const errors: Record<string, string> = {};

    // 이메일 검증
    if (!formData.email) {
      errors.email = '이메일은 필수입니다';
    } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(formData.email)) {
      errors.email = '올바른 이메일 형식이 아닙니다';
    }

    // 비밀번호 검증
    if (!formData.password) {
      errors.password = '비밀번호는 필수입니다';
    } else if (formData.password.length < 6) {
      errors.password = '비밀번호는 최소 6자 이상이어야 합니다';
    }

    // 비밀번호 확인
    if (formData.password !== formData.confirmPassword) {
      errors.confirmPassword = '비밀번호가 일치하지 않습니다';
    }

    // 이름 검증
    if (!formData.name) {
      errors.name = '이름은 필수입니다';
    } else if (formData.name.length < 2) {
      errors.name = '이름은 최소 2자 이상이어야 합니다';
    }

    setFormError(errors);
    return Object.keys(errors).length === 0;
  };

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;
    setFormData((prev) => ({
      ...prev,
      [name]: value,
    }));
    if (formError[name]) {
      setFormError((prev) => ({
        ...prev,
        [name]: '',
      }));
    }
  };

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();

    if (!validateForm()) {
      return;
    }

    try {
      await register(
        formData.email,
        formData.password,
        formData.name,
        formData.tel || undefined
      );
      navigate('/projects');
    } catch (err) {
      console.error('회원가입 오류:', err);
    }
  };

  return (
    <div className="auth-page">
      <div className="auth-form">
        <h1 className="auth-form__title">회원가입</h1>

        {error && <div className="auth-form__error">{error}</div>}

        <form onSubmit={handleSubmit} className="auth-form__form">
          <div className="auth-form__field">
            <label htmlFor="email" className="auth-form__label">
              이메일
            </label>
            <input
              type="email"
              id="email"
              name="email"
              value={formData.email}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.email ? 'auth-form__input--error' : ''
              }`}
              placeholder="user@example.com"
              disabled={isLoading}
            />
            {formError.email && (
              <span className="auth-form__error-message">{formError.email}</span>
            )}
          </div>

          <div className="auth-form__field">
            <label htmlFor="name" className="auth-form__label">
              이름
            </label>
            <input
              type="text"
              id="name"
              name="name"
              value={formData.name}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.name ? 'auth-form__input--error' : ''
              }`}
              placeholder="홍길동"
              disabled={isLoading}
            />
            {formError.name && (
              <span className="auth-form__error-message">{formError.name}</span>
            )}
          </div>

          <div className="auth-form__field">
            <label htmlFor="tel" className="auth-form__label">
              전화번호 (선택)
            </label>
            <input
              type="tel"
              id="tel"
              name="tel"
              value={formData.tel}
              onChange={handleChange}
              className="auth-form__input"
              placeholder="010-1234-5678"
              disabled={isLoading}
            />
          </div>

          <div className="auth-form__field">
            <label htmlFor="password" className="auth-form__label">
              비밀번호
            </label>
            <input
              type="password"
              id="password"
              name="password"
              value={formData.password}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.password ? 'auth-form__input--error' : ''
              }`}
              placeholder="••••••"
              disabled={isLoading}
            />
            {formError.password && (
              <span className="auth-form__error-message">
                {formError.password}
              </span>
            )}
          </div>

          <div className="auth-form__field">
            <label htmlFor="confirmPassword" className="auth-form__label">
              비밀번호 확인
            </label>
            <input
              type="password"
              id="confirmPassword"
              name="confirmPassword"
              value={formData.confirmPassword}
              onChange={handleChange}
              className={`auth-form__input ${
                formError.confirmPassword ? 'auth-form__input--error' : ''
              }`}
              placeholder="••••••"
              disabled={isLoading}
            />
            {formError.confirmPassword && (
              <span className="auth-form__error-message">
                {formError.confirmPassword}
              </span>
            )}
          </div>

          <button
            type="submit"
            className="auth-form__button"
            disabled={isLoading}
          >

            </Link>
          </p>
        </div>
      </div>
    </div>
  );
};

export default RegisterPage;
```
