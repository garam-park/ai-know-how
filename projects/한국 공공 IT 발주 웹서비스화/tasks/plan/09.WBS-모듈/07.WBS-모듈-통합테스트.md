# WBS 모듈 통합 테스트

## 예상 소요시간
| 구분 | 시간 |
|------|------|
| 사람 (숙련 개발자) | ~30분 |
| AI 에이전트 | ~10분 |

## What (무엇을)
Jest와 supertest를 사용하여 WBS 모듈의 전체 시나리오에 대한 통합 테스트를 작성합니다.

## Why (왜)
트리 구조 + 진척률 재계산 + Scope 필터링이 복합적으로 동작하므로, 각 기능이 독립적으로뿐만 아니라 함께 정상 작동하는지 검증해야 합니다.

- `09.WBS-모듈/06.WBS-정렬-이동-API.md`

## 수행 가이드

### Step 1: 테스트 파일 생성
`apps/backend/src/modules/wbs/__tests__/wbs.integration.spec.ts`:
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { INestApplication } from '@nestjs/common';
import * as request from 'supertest';
import { WbsModule } from '../wbs.module';
import { PrismaService } from 'src/prisma/prisma.service';

describe('WBS Integration Tests', () => {
  let app: INestApplication;
  let prisma: PrismaService;
  let projectId: number;
  let categoryId1: number;
  let categoryId2: number;

  beforeAll(async () => {
    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [WbsModule],
      providers: [PrismaService],
    }).compile();

    app = moduleFixture.createNestApplication();
    await app.init();

    prisma = moduleFixture.get<PrismaService>(PrismaService);

    // 테스트용 프로젝트 생성
    const project = await prisma.project.create({
      data: {
        name: 'Test Project',
        maxWbsDepth: 5,
      },
    });
    projectId = project.id;
  });

  afterAll(async () => {
    await prisma.$disconnect();
    await app.close();
  });

  describe('시나리오 1: CATEGORY/TASK 생성', () => {
    it('CATEGORY 3개 생성', async () => {
      const response = await request(app.getHttpServer())
        .post(`/api/v1/projects/${projectId}/wbs-nodes`)
        .send({
          type: 'CATEGORY',
          title: 'Category 1',
          code: 'CAT-1',
        })
        .expect(201);

      expect(response.body.code).toBe(201000);
      categoryId1 = response.body.result.node.id;
    });

    it('TASK 6개 생성 (3개씩 2개 CATEGORY 하위)', async () => {
      for (let i = 1; i <= 3; i++) {
        await request(app.getHttpServer())
          .post(`/api/v1/projects/${projectId}/wbs-nodes`)
          .send({
            parentId: categoryId1,
            type: 'TASK',
            title: `Task 1-${i}`,
            weight: i,
          })
          .expect(201);
      }
    });
  });

  describe('시나리오 2: TASK 진척률 변경 및 가중평균 재계산', () => {
    it('TASK 진척률 50 → 80 변경', async () => {
      const tasks = await prisma.wbsNode.findMany({
        where: {
          projectId,
          type: 'TASK',
          parentId: categoryId1,
        },
      });

      const taskId = tasks[0].id;

      const response = await request(app.getHttpServer())
        .patch(`/api/v1/projects/${projectId}/wbs-nodes/${taskId}/progress`)
        .send({ progress: 80 })
        .expect(200);

      expect(response.body.code).toBe(200000);
      expect(response.body.result.node.progress).toBe(80);
    });

    it('CATEGORY 진척률이 자동 재계산되어야 함 (가중평균)', async () => {
      const response = await request(app.getHttpServer())
        .get(`/api/v1/projects/${projectId}/wbs-nodes/${categoryId1}`)
        .expect(200);

      const category = response.body.result;
      expect(category.progress).toBeGreaterThan(0);
      expect(category.progress).toBeLessThanOrEqual(100);
    });
  });

  describe('시나리오 3: weight 다르게 설정 시 가중평균', () => {
    it('weight 1,2,3 일 때 정확한 가중평균 계산', async () => {
      const tasks = await prisma.wbsNode.findMany({
        where: {
          projectId,
          type: 'TASK',
          parentId: categoryId1,
        },
        orderBy: { createdAt: 'asc' },
      });

      // Task 1: progress=80, weight=1
      // Task 2: progress=0, weight=2
      // Task 3: progress=0, weight=3
      // Expected: (80*1 + 0*2 + 0*3) / (1+2+3) = 80/6 ≈ 13

      // Task 2 진척률 업데이트 (100%)
      await request(app.getHttpServer())
        .patch(`/api/v1/projects/${projectId}/wbs-nodes/${tasks[1].id}/progress`)
        .send({ progress: 100 })
        .expect(200);

      // Category 확인
      const catResponse = await request(app.getHttpServer())
        .get(`/api/v1/projects/${projectId}/wbs-nodes`)
        .query({ scope: 'ALL' })
        .expect(200);

      const category = catResponse.body.result.nodes.find(
        (n) => n.id === categoryId1,
      );
      // (80*1 + 100*2 + 0*3) / 6 = 280/6 ≈ 47
      expect(category.progress).toBe(47);
    });
  });

  describe('시나리오 4: Scope=ALL (주관사PM)', () => {
    it('전체 WBS 트리 반환', async () => {
      const response = await request(app.getHttpServer())
        .get(`/api/v1/projects/${projectId}/wbs-nodes`)
        .query({ scope: 'ALL' })
        .expect(200);

      expect(response.body.code).toBe(200000);
      expect(response.body.result.nodes.length).toBeGreaterThan(0);
      expect(response.body.result.totalCount).toBeGreaterThan(0);
    });
  });

  describe('시나리오 5: Scope=OWN_COMPANY (참여사PM)', () => {
    it('자기 회사 노드 + 경로만 포함', async () => {
      // 테스트용 회사와 사용자 생성 필요
      const response = await request(app.getHttpServer())
        .get(`/api/v1/projects/${projectId}/wbs-nodes`)
        .query({ scope: 'OWN_COMPANY' })
        .set('Authorization', 'Bearer test-token-company')
        .expect(200);

      expect(response.body.code).toBe(200000);
      // 조회된 노드가 회사 필터링된 것인지 검증
    });
  });

  describe('시나리오 6: 노드 이동 및 depth 갱신', () => {
    it('노드 이동 시 depth 자동 갱신', async () => {
      const response = await request(app.getHttpServer())
        .patch(`/api/v1/projects/${projectId}/wbs-nodes/reorder`)
        .send({
          orders: [
            {
              nodeId: categoryId1,
              order: 1,
              parentId: null,
            },
          ],
        })
        .expect(200);

      expect(response.body.code).toBe(200000);
      const updated = response.body.result.nodes[0];
      expect(updated.depth).toBe(0);
    });

    it('이동 후 이전/새 부모의 진척률 재계산', async () => {
      // 이동 전후 진척률 비교
      const before = await request(app.getHttpServer())
        .get(`/api/v1/projects/${projectId}/wbs-nodes`)
        .expect(200);

      const response = await request(app.getHttpServer())
        .patch(`/api/v1/projects/${projectId}/wbs-nodes/reorder`)
        .send({
          orders: [
            {
              nodeId: categoryId1,
              order: 2,
            },
          ],
        })
        .expect(200);

      expect(response.body.code).toBe(200000);
    });
  });

  describe('시나리오 7: 노드 삭제 및 진척률 재계산', () => {
    it('노드 삭제 시 하위 노드도 soft delete', async () => {
      const tasks = await prisma.wbsNode.findMany({
        where: {
          projectId,
          type: 'TASK',
          parentId: categoryId1,
          deletedAt: null,
        },
      });

      const taskToDelete = tasks[0];

      const response = await request(app.getHttpServer())
        .delete(`/api/v1/projects/${projectId}/wbs-nodes/${taskToDelete.id}`)
        .expect(200);

      expect(response.body.code).toBe(200000);

      // 삭제 확인
      const deleted = await prisma.wbsNode.findUnique({
        where: { id: taskToDelete.id },
      });
      expect(deleted.deletedAt).not.toBeNull();
    });

    it('삭제 후 상위 CATEGORY 진척률 재계산', async () => {
      // 삭제 후 CATEGORY의 진척률이 재계산되었는지 확인
      const category = await prisma.wbsNode.findUnique({
        where: { id: categoryId1 },
      });

      expect(category.progress).toBeDefined();
    });
  });
});
```
