# 대시보드 집계 API

## 예상 소요시간
| 구분 | 시간 |
|------|------|
| 사람 (숙련 개발자) | ~40분 |

## What (무엇을)
GET /api/v1/projects/:id/dashboard 엔드포인트를 구현하여 프로젝트의 종합 현황 데이터를 반환합니다.

## Why (왜)
프로젝트 전체 현황을 한눈에 보여주는 핵심 화면의 데이터를 제공합니다. 프로젝트 정보, 전체 진척률, 컨소시엄사별 진척률, 멤버 통계, WBS 노드 상태를 집계합니다.

### Step 1: 대시보드 집계 유틸리티 함수 추가
`apps/backend/src/modules/dashboard/utils/dashboard.utils.ts`:
```typescript
import { DashboardResponse } from '../dto/dashboard.dto';

export function calculateDaysLeft(endDate: Date): number {
  const today = new Date();
  const end = new Date(endDate);
  const diffTime = end.getTime() - today.getTime();
  const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  return diffDays;
}

export function calculateOverallProgress(
  nodes: any[],
): number {
  // 최상위 WBS 노드들(parentId === null)의 가중평균
  const rootNodes = nodes.filter(n => n.parentId === null && n.deletedAt === null);
  
  if (rootNodes.length === 0) return 0;

  const totalWeight = rootNodes.reduce((sum, n) => sum + n.weight, 0);
  const weightedSum = rootNodes.reduce(
    (sum, n) => sum + n.progress * n.weight,
    0,
  );

  return totalWeight > 0 ? Math.round(weightedSum / totalWeight) : 0;
}
```

### Step 2: DashboardService에 집계 로직 추가
`apps/backend/src/modules/dashboard/dashboard.service.ts`:
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from 'src/prisma/prisma.service';
import { DashboardResponse } from './dto/dashboard.dto';
import { calculateDaysLeft, calculateOverallProgress } from './utils/dashboard.utils';

@Injectable()
export class DashboardService {
  constructor(private prisma: PrismaService) {}

  async getDashboard(projectId: number, scope: string, userId: number): Promise<DashboardResponse> {
    // 1. 프로젝트 기본 정보 조회
    const project = await this.prisma.project.findUnique({
      where: { id: projectId },
    });

    if (!project) {
      throw new NotFoundException('Project not found');
    }

    const daysLeft = calculateDaysLeft(project.endDate);

    // 2. WBS 노드 조회 (Scope 적용)
    let wbsNodes: any[] = [];
    
    if (scope === 'ALL') {
      wbsNodes = await this.prisma.wbsNode.findMany({
        where: {
          projectId,
          deletedAt: null,
        },
        include: {
          projectCompany: true,
        },
      });
    } else if (scope === 'OWN_COMPANY') {
      // 사용자 소속 회사의 노드만
      const user = await this.prisma.user.findUnique({
        where: { id: userId },
        include: { company: true },
      });

      wbsNodes = await this.prisma.wbsNode.findMany({
        where: {
          projectId,
          projectCompanyId: user.company.id,
          deletedAt: null,
        },
        include: {
          projectCompany: true,
        },
      });
    } else if (scope === 'OWN_TASK') {
      wbsNodes = await this.prisma.wbsNode.findMany({
        where: {
          projectId,
          assigneeId: userId,
          deletedAt: null,
        },
        include: {
          projectCompany: true,
        },
      });
    }

    // 3. 전체 진척률 계산
    const overallProgress = calculateOverallProgress(wbsNodes);

    // 4. 컨소시엄사별 진척률 계산
    const progressByCompany = await this.calculateProgressByCompany(
      projectId,
      scope,
      userId,
    );

    // 5. 프로젝트 멤버 통계
    const members = await this.calculateMembers(projectId, scope, userId);

    // 6. WBS 노드 상태 통계
    const wbsSummary = this.calculateWbsSummary(wbsNodes);

    return {
      project: {
        id: project.id,
        name: project.name,
        startDate: project.startDate,
        endDate: project.endDate,
        daysLeft,
      },
      progress: {
        overall: overallProgress,
        byCompany: progressByCompany,
      },
      members: {
        total: members.total,
        byCompany: members.byCompany,
      },
      wbs: wbsSummary,
    };
  }

  private async calculateProgressByCompany(
    projectId: number,
    scope: string,
    userId: number,
  ): Promise<any[]> {
    // 프로젝트의 모든 컨소시엄사 조회
    const projectCompanies = await this.prisma.projectCompany.findMany({
      where: { projectId },
      include: { company: true },
    });

    const result: any[] = [];

    for (const pc of projectCompanies) {
      // 각 회사별 WBS 노드 조회 (root 노드만)
      let companyNodes = await this.prisma.wbsNode.findMany({
        where: {
          projectId,
          projectCompanyId: pc.id,
          deletedAt: null,
        },
      });

      // Scope이 OWN_COMPANY인 경우 필터링
      if (scope === 'OWN_COMPANY') {
        const user = await this.prisma.user.findUnique({
          where: { id: userId },
          include: { company: true },
        });

        if (pc.companyId !== user.company.id) {
          continue; // 다른 회사는 스킵
        }
      }

      const companyProgress = calculateOverallProgress(companyNodes);

      result.push({
        company: {
          id: pc.company.id,
          name: pc.company.name,
        },
        progress: companyProgress,
      });
    }

    return result;
  }

  private async calculateMembers(
    projectId: number,
    scope: string,
    userId: number,
  ): Promise<{ total: number; byCompany: any[] }> {
    // 프로젝트 멤버 조회
    let memberQuery: any = {
      projectId,
    };

    if (scope === 'OWN_COMPANY') {
      const user = await this.prisma.user.findUnique({
        where: { id: userId },
        include: { company: true },
      });

      memberQuery.companyId = user.company.id;
    }

    const members = await this.prisma.projectMember.findMany({
      where: memberQuery,
      include: { company: true },
    });

    const total = members.length;

    // 회사별 멤버 수 집계
    const byCompanyMap = new Map<number, { id: number; name: string; count: number }>();

    for (const member of members) {
      const key = member.company.id;
      if (!byCompanyMap.has(key)) {
        byCompanyMap.set(key, {
          id: member.company.id,
          name: member.company.name,
          count: 0,
        });
      }
      const data = byCompanyMap.get(key)!;
      data.count += 1;
    }

    return {
      total,
      byCompany: Array.from(byCompanyMap.values()).map(item => ({
        company: {
          id: item.id,
          name: item.name,
        },
        count: item.count,
      })),
    };
  }

  private calculateWbsSummary(nodes: any[]): any {
    let total = 0;
    let completed = 0;
    let inProgress = 0;
    let notStarted = 0;

    for (const node of nodes) {
      if (node.type === 'TASK') {
        total += 1;
        if (node.progress === 100) {
          completed += 1;
        } else if (node.progress > 0) {
          inProgress += 1;
        } else {
          notStarted += 1;
        }
      }
    }

    return {
      total,
      completed,
      inProgress,
      notStarted,
    };
  }
}
```

### Step 3: DashboardController에 집계 엔드포인트 추가
`apps/backend/src/modules/dashboard/dashboard.controller.ts`:
```typescript
import { Controller, Get, Param, Request } from '@nestjs/common';
import { DashboardService } from './dashboard.service';
import { RequirePermission } from 'src/auth/decorators/require-permission.decorator';
import { Resource, Action } from 'src/auth/types/permission.types';

@Controller('api/v1/projects/:id/dashboard')
export class DashboardController {
  constructor(private dashboardService: DashboardService) {}

  @Get()
  @RequirePermission(Resource.PROJECT, Action.VIEW)
  async getDashboard(
    @Param('id') projectId: string,
    @Request() req,
  ) {
    const scope = req.scope; // PermissionGuard에서 설정
    const userId = req.user.id;

    const result = await this.dashboardService.getDashboard(
      parseInt(projectId),
      scope,
      userId,
    );

    return {
      code: 200000,
      result,
    };
  }
}
```
