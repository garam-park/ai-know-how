# 프로젝트 생성 화면

## 예상 소요시간
| 구분 | 시간 |

## What (무엇을)
프로젝트 생성 페이지에서 프로젝트명, 코드, 설명, 시작일, 종료일, 예산 등을 입력받아 새 사업을 등록합니다. 날짜 입력은 DatePicker를 사용하며, 성공 후 프로젝트 목록으로 리다이렉트됩니다.

## Why (왜)
새로운 사업을 시스템에 등록하기 위한 진입점입니다.

### Step 2: ProjectCreatePage 컴포넌트 생성
`apps/frontend/src/pages/projects/ProjectCreatePage.tsx` 파일을 생성합니다:

```typescript
import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import DatePicker from 'react-datepicker';
import { projectService } from '../../services/projectService';
import Button from '../../components/Button';
import Input from '../../components/Input';
import Modal from '../../components/Modal';
import Toast from '../../components/Toast';
import 'react-datepicker/dist/react-datepicker.css';
import './ProjectCreatePage.css';

const ProjectCreatePage: React.FC = () => {
  const navigate = useNavigate();
  const [isOpen, setIsOpen] = useState(true);
  const [loading, setLoading] = useState(false);
  const [toast, setToast] = useState<{
    message: string;
    type: 'success' | 'error';
  } | null>(null);

  const [formData, setFormData] = useState({
    name: '',
    code: '',
    description: '',
    startDate: new Date(),
    endDate: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000), // 30일 후
    budget: '',
  });

  const [errors, setErrors] = useState<Record<string, string>>({});

  const validateForm = (): boolean => {
    const newErrors: Record<string, string> = {};

    if (!formData.name.trim()) {
      newErrors.name = '프로젝트명은 필수입니다';
    }

    if (formData.startDate >= formData.endDate) {
      newErrors.dates = '종료일은 시작일보다 뒤여야 합니다';
    }

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleInputChange = (
    e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
  ) => {
    const { name, value } = e.target;
    setFormData((prev) => ({
      ...prev,
      [name]: value,
    }));
    if (errors[name]) {
      setErrors((prev) => ({
        ...prev,
        [name]: '',
      }));
    }
  };

  const handleDateChange = (date: Date | null, field: 'startDate' | 'endDate') => {
    if (date) {
      setFormData((prev) => ({
        ...prev,
        [field]: date,
      }));
      if (errors.dates) {
        setErrors((prev) => ({
          ...prev,
          dates: '',
        }));
      }
    }
  };

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();

    if (!validateForm()) {
      return;
    }

    try {
      setLoading(true);
      const payload = {
        name: formData.name,
        code: formData.code || undefined,
        description: formData.description || undefined,
        startDate: formData.startDate.toISOString().split('T')[0],
        endDate: formData.endDate.toISOString().split('T')[0],
        budget: formData.budget ? parseInt(formData.budget, 10) : undefined,
      };

      await projectService.createProject(payload);
      setToast({ message: '프로젝트가 생성되었습니다', type: 'success' });

      setTimeout(() => {
        navigate('/projects');
      }, 1500);
    } catch (err) {
      setToast({
        message: err instanceof Error ? err.message : '프로젝트 생성 실패',
        type: 'error',
      });
    } finally {
      setLoading(false);
    }
  };

  const handleClose = () => {
    navigate('/projects');
  };

  return (
    <>
      <Modal
        isOpen={isOpen}
        title="프로젝트 생성"
        onClose={handleClose}
        size="large"
      >
        <form onSubmit={handleSubmit} className="project-create__form">
          <div className="project-create__field">
            <Input
              label="프로젝트명 *"
              type="text"
              name="name"
              value={formData.name}
              onChange={handleInputChange}
              placeholder="예: 2024년 정부부처 시스템 구축"
              error={errors.name}
            />
          </div>

          <div className="project-create__field">
            <Input
              label="프로젝트 코드"
              type="text"
              name="code"
              value={formData.code}
              onChange={handleInputChange}
              placeholder="예: PRJ-2024-001"
            />
          </div>

          <div className="project-create__field">
            <label className="project-create__label">설명</label>
            <textarea
              name="description"
              value={formData.description}
              onChange={handleInputChange}
              placeholder="프로젝트 설명을 입력하세요"
              className="project-create__textarea"
              rows={4}
            />
          </div>

          <div className="project-create__row">
            <div className="project-create__field">
              <label className="project-create__label">시작일 *</label>
              <DatePicker
                selected={formData.startDate}
                onChange={(date) => handleDateChange(date, 'startDate')}
                dateFormat="yyyy-MM-dd"
                className="project-create__datepicker"
                minDate={new Date()}
              />
            </div>

            <div className="project-create__field">
              <label className="project-create__label">종료일 *</label>
              <DatePicker
                selected={formData.endDate}
                onChange={(date) => handleDateChange(date, 'endDate')}
                dateFormat="yyyy-MM-dd"
                className="project-create__datepicker"
                minDate={formData.startDate}
              />
            </div>
          </div>

          {errors.dates && (
            <div className="project-create__error">{errors.dates}</div>
          )}

          <div className="project-create__field">
            <Input
              label="예산"
              type="number"
              name="budget"
              value={formData.budget}
              onChange={handleInputChange}
              placeholder="예: 100000000"
            />
          </div>

          <div className="project-create__footer">
            <Button
              variant="secondary"
              onClick={handleClose}
              disabled={loading}
            >
              취소
            </Button>
            <Button variant="primary" type="submit" loading={loading}>
              생성
            </Button>
          </div>
        </form>
      </Modal>

      {toast && (
        <Toast
          message={toast.message}
          type={toast.type}
          onClose={() => setToast(null)}
        />
      )}
    </>
  );
};

export default ProjectCreatePage;
```
